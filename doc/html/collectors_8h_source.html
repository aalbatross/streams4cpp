567 static auto toMap(KeyMapper &&keyMapper, ValueMapper &&valueMapper, MergeFunction &&mergeFunction) {
568 return streams::Collector{[] { return std::vector<T>(); },
569 [](std::vector<T> &intermediate, const auto &element) {
570 intermediate.emplace_back(element);
571 },
572 [keyMapper, valueMapper, mergeFunction](std::vector<T> &intermediate) {
573 using K = typename std::invoke_result<KeyMapper, T>::type;
574 using V = typename std::invoke_result<ValueMapper, T>::type;
575
576 std::map<K, std::vector<V>> iResult;
577 for (const T &item : intermediate) {
578 iResult.try_emplace(keyMapper(item), std::vector<V>());
579 iResult[keyMapper(item)].emplace_back(valueMapper(item));
580 }
581
582 std::map<K, V> result;
583 for (const auto &item : iResult) {
584 result[item.first] = std::reduce(item.second.begin(), item.second.end(), V{}, mergeFunction);
585 }
