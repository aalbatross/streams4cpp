148 template<typename Discriminator>
149 auto groupedBy(Discriminator &&discriminator) {
150 auto vec = toVector();
151 using K = typename std::invoke_result<Discriminator, T>::type;
152 std::unordered_map<K, std::vector<T>> output;
153 for (auto element : vec) {
154 K key = discriminator(element);
155 output.try_emplace(key, std::vector<T>());
156 output[key].emplace_back(element);
157 }
158 return output;
159 }
