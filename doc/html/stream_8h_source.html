165 template<typename Discriminator>
166 auto groupedBy(Discriminator &&discriminator) {
167 auto vec = toVector();
168 using K = typename std::invoke_result<Discriminator, T>::type;
169 std::unordered_map<K, std::vector<T>> output;
170 for (auto element : vec) {
171 K key = discriminator(element);
172 output.try_emplace(key, std::vector<T>());
173 output[key].emplace_back(element);
174 }
175 return output;
176 }
